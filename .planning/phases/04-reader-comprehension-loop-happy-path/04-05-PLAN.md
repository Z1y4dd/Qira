```typescript
import { readFileSync } from 'node:fs';
import { join } from 'node:path';
import { globSync } from 'tinyglobby';
import { describe, expect, test } from 'vitest';
```

The test globs all layout and page files under `src/app/(authenticated)/**` and asserts each contains `export const dynamic = 'force-dynamic'`.

```typescript
const REPO_ROOT = process.cwd();

const layoutsAndPages = globSync(
  ['src/app/(authenticated)/**/{layout,page}.tsx'],
  { cwd: REPO_ROOT, ignore: ['**/node_modules/**', '**/.next/**'] }
);

describe('force-dynamic coverage', () => {
  test('source tree is non-empty (sanity)', () => {
    expect(layoutsAndPages.length).toBeGreaterThan(0);
  });

  for (const file of layoutsAndPages) {
    test(`${file} exports dynamic = 'force-dynamic'`, () => {
      const contents = readFileSync(join(REPO_ROOT, file), 'utf8');
      expect(
        contents,
        `${file} is missing \`export const dynamic = 'force-dynamic'\`. ` +
          'All authenticated pages and layouts require force-dynamic to prevent ' +
          'stale RSC cache from leaking session data across users (Supabase SSR pitfall).',
      ).toContain("export const dynamic = 'force-dynamic'");
    });
  }
});
```

Note: the `(placement-gate)/layout.tsx` already exports `force-dynamic` (confirmed at line 17 of the file). The new `reader/layout.tsx` (Plan 03), `reader/[attemptId]/page.tsx` (Plan 03), and `reader/[attemptId]/result/page.tsx` (Plan 05 Task 5.1) must all also have it — verify each has the export before running this test. The `reader/page.tsx` auto-route page also needs it — add if missing.
