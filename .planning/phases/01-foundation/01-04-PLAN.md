- **Task 4.2 — Author `src/lib/zod.ts` `ArabicText` schema and `src/db/normalize.ts` `nfc()` helper**
  - **Files:** `src/lib/zod.ts`, `src/db/normalize.ts`
  - **Action:**
    - `pnpm add zod@^4`.
    - Create `src/lib/zod.ts` per RESEARCH §D:
      ```ts
      import { z } from 'zod';

      /**
       * Validates that Arabic text is NFC-normalized BEFORE persistence.
       * Use as the schema for every Server Action / Route Handler input field
       * that accepts Arabic content (titleAr, bodyAr, promptAr, choices.textAr).
       *
       * Locked decision (RESEARCH §I-6): NFC, not NFKC. NFKC can destroy
       * intentional Tashkeel-bearing presentation forms — silent data loss.
       */
      export const ArabicText = z.string().min(1).refine(
        (s) => s.normalize('NFC') === s,
        { message: 'Arabic text must be NFC-normalized' }
      );
      ```
    - Create `src/db/normalize.ts` per RESEARCH §D:
      ```ts
      /**
       * Belt-and-suspenders NFC normalizer applied at the DB write boundary
       * inside Service Layer functions, AFTER Zod validation has run.
       * Together with the Zod `ArabicText` schema, the invariant
       * "any Arabic string stored in Postgres is NFC-normalized" holds.
       */
      export function nfc<T extends Record<string, unknown>>(obj: T, fields: (keyof T)[]): T {
        const out = { ...obj };
        for (const f of fields) {
          if (typeof out[f] === 'string') {
            (out as Record<string, unknown>)[f] = (out[f] as string).normalize('NFC');
          }
        }
        return out;
      }
      ```
  - **Verification:** `pnpm tsc --noEmit` passes. `node --input-type=module -e "import('./src/lib/zod.ts').then(m => console.log(m.ArabicText.parse('السلام')))"` succeeds (tsx may be required to execute TS directly — alternatively, write a temporary `.mjs` shim or just rely on the unit test in Task 4.5).

      ```ts
      import { ArabicText } from '@/lib/zod';
      import { describe, test, expect } from 'vitest';

      describe('ArabicText Zod schema', () => {
        test('accepts NFC-normalized Arabic', () => {
          const ok = 'السلام عليكم'.normalize('NFC');
          expect(() => ArabicText.parse(ok)).not.toThrow();
        });
        test('rejects empty string', () => {
          expect(() => ArabicText.parse('')).toThrow();
        });
        test('rejects non-NFC (NFD-form) input', () => {
          // ARABIC LETTER ALEF (U+0627) + ARABIC FATHA (U+064E) — decomposed sequence
          const nfd = 'ا' + 'َ';
          // Verify the input is indeed NFD vs NFC for safety
          if (nfd === nfd.normalize('NFC')) {
            // Picked codepoints that happen to be NFC-stable; pick a different pair.
            return;
          }
          expect(() => ArabicText.parse(nfd)).toThrow(/NFC-normalized/);
        });
      });
      ```

      ```ts
      import { describe, test, expect } from 'vitest';

        test('normalizes specified Arabic fields', () => {
          const input = { titleAr: 'الكتاب', bodyAr: 'الجملة', wordCount: 2 };
          const out = nfc(input, ['titleAr', 'bodyAr']);
      ```
